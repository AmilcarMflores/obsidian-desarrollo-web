## if - if else
```javascript
let rain = true;

if (rain) {
    console.log("** Taking my umbrella when I need to go outside **");
} else {
    console.log("** I can leave my umbrella at home **");
}
```

```javascript
let age = 25;

if (age < 18) {
    console.log("We're very sorry, but you can't get in under 18");
} else {
    console.log("Welcome!");
}
```
## else if
```javascript
let age = 10;
let cost = 0;
let message;

if  (age < 3) {
    cost = 0;
    message = "Access is free under three";
} else if (age >= 3 && age < 12) {
    cost = 5;
    message = "With the child discount, the fee is 5 dollars";
} else if (age >= 12 && age < 65) {
    cost = 10;
    message = "A regular ticket costs 10 dollars";
} else {
    cost = 7;
    message = "A ticket is 7 dollars."
}

console.log(message);
console.log("Your total cost " + cost);
```
## Operadores ternarios condicionales

`expresion? declaración-para-verdadero : declaración-asociada-como-falso;`

```javascript
let age = 20;
let access = age < 18 ? "denied" : "allowed";
console.log(`Access is ${access}.`);
```
## Switch
```javascript
let activity = "Drive to work";

switch (activity) {
    case "Get up":
        console.log("It is 6:30AM");
        break;
    case "Breakfast":
        console.log("It is 7:00AM");
        break;
    case "Drive to work":
        console.log("It is 8:00AM");
        break;
    case "Lunch":
        console.log("It is 12:00PM");
        break;
    case "Drive home":
        console.log("It is 5:00PM");
        break;
    case "Dinner":
        console.log("It is 6:30PM");
        break;
}   
```
## The default case
En un switch hay una etiqueta especial llamada `default`, su funcionamiento que es similar al `else`:
```javascript
let activity = "Sleep";

switch (activity) {
    case "Get up":
        console.log("It is 6:30AM");
        break;
    case "Breakfast":
        console.log("It is 7:00AM");
        break;
    case "Drive to work":
        console.log("It is 8:00AM");
        break;
    case "Lunch":
        console.log("It is 12:00PM");
        break;
    case "Drive home":
        console.log("It is 5:00PM");
        break;
    case "Dinner":
        console.log("It is 6:30PM");
        break;
    default:
        console.log("I cannot determine the current time.");
        break;
}   
```
